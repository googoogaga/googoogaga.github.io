One more change to %net.c:

GOOFUNC(close_socket) (P sockfd)
{
#if defined(_MSC_VER)
  closesocket(sockUnwrap(sockfd));
#else
  close(sockUnwrap(sockfd));
#endif

  return YPtrue;
}
