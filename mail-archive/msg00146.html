  ;;; JUL 19, 2002
  ;;; 
  ;;; VERSION = 0.140
  ;;; ROOT    = /tmp/goo/lib/goo/

  goo/user 0<= (df foo () "foo")
  goo/user 0=> #{Met foo ()}
  goo/user 0<= (foo)
  goo/user 0=> "foo"

  goo/user 0=> #{Met foo ()}
